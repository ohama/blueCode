```fsharp
let private extractQuotedContent (command: string) : string * string * string =
    // Walk character-by-character. Track quote state (single vs double).
    // Return (withDoubleQuotes, fullyUnquoted, unquotedKeepQuoteChars):
    //   withDoubleQuotes: outside single quotes (double-quoted content preserved)
    //   fullyUnquoted: outside both single AND double quotes
    //   unquotedKeepQuoteChars: like fullyUnquoted but preserves quote chars
    //
    // Port exactly from bash_security.py extract_quoted_content. Pure string
    // traversal — no regex. Handles:
    //   \\ escapes (backslash)
    //   '...' single quotes (suppress all special chars)
    //   "..." double quotes (suppress most but not $, `, \)
    //
    // Implementation sketch (adapt from Python, keep logic identical):
    let sbWithDq   = System.Text.StringBuilder()
    let sbFully    = System.Text.StringBuilder()
    let sbKeep     = System.Text.StringBuilder()
    let mutable i = 0
    let mutable inSingle = false
    let mutable inDouble = false
    while i < command.Length do
        let c = command.[i]
        // Backslash escape
        if c = '\\' && i + 1 < command.Length && not inSingle then
            // Copy both chars verbatim into all three outputs
            // (Python behaviour: backslash + next char preserved).
            sbWithDq.Append(c)  |> ignore
            sbWithDq.Append(command.[i + 1]) |> ignore
            if not inDouble then
                sbFully.Append(c) |> ignore
                sbFully.Append(command.[i + 1]) |> ignore
                sbKeep.Append(c)  |> ignore
                sbKeep.Append(command.[i + 1]) |> ignore
            i <- i + 2
        elif c = '\'' && not inDouble then
            inSingle <- not inSingle
            sbKeep.Append(c) |> ignore
            i <- i + 1
        elif c = '"' && not inSingle then
            inDouble <- not inDouble
            sbWithDq.Append(c) |> ignore
            sbKeep.Append(c)   |> ignore
            i <- i + 1
        else
            // Not in single quote -> contributes to withDoubleQuotes
            if not inSingle then
                sbWithDq.Append(c) |> ignore
            // Not in any quote -> contributes to fullyUnquoted and keepQuoteChars
            if not inSingle && not inDouble then
                sbFully.Append(c) |> ignore
                sbKeep.Append(c)  |> ignore
            i <- i + 1
    sbWithDq.ToString(), sbFully.ToString(), sbKeep.ToString()

let private stripSafeRedirections (content: string) : string =
    // Strip 2>&1, >/dev/null, </dev/null before dangerous-variable checks.
    let mutable result = content

let private hasUnescapedChar (content: string) (ch: char) : bool =
    // Count consecutive backslashes before each occurrence of ch.
    // Even count (including zero) = NOT escaped.
    let mutable i = 0
    let mutable found = false
    while i < content.Length && not found do
        if content.[i] = ch then
            // Count preceding backslashes
            let mutable back = 0
            let mutable j = i - 1
            while j >= 0 && content.[j] = '\\' do
                back <- back + 1
                j    <- j - 1
            if back % 2 = 0 then found <- true
        i <- i + 1
    found

let private isEscapedAtPosition (content: string) (pos: int) : bool =
    let mutable back = 0
    let mutable j = pos - 1
    while j >= 0 && content.[j] = '\\' do
        back <- back + 1
        j    <- j - 1
    back % 2 = 1

let private splitCommand (command: string) : string list =
    // Split compound commands on ;, ||, |, && respecting quotes.
    // Single & (background) does NOT split.
    // Port exactly from bash_security.py split_command.
    let segments = System.Collections.Generic.List<string>()
    let current  = System.Text.StringBuilder()
    let mutable inSingle = false
    let mutable inDouble = false
    let mutable i = 0
    let flush () =
        let s = current.ToString().Trim()
        if s <> "" then segments.Add(s)
        current.Clear() |> ignore
    while i < command.Length do
        let c = command.[i]
        if c = '\'' && not inDouble then
            inSingle <- not inSingle
            current.Append(c) |> ignore
            i <- i + 1
        elif c = '"' && not inSingle then
            inDouble <- not inDouble
            current.Append(c) |> ignore
            i <- i + 1
        elif not inSingle && not inDouble then
            // && / ||
            if i + 1 < command.Length && (command.[i] = '&' && command.[i + 1] = '&'
                                          || command.[i] = '|' && command.[i + 1] = '|') then
                flush ()
                i <- i + 2
            elif c = ';' || c = '|' then
                flush ()
                i <- i + 1
            else
                current.Append(c) |> ignore
                i <- i + 1
        else
            current.Append(c) |> ignore
            i <- i + 1
    flush ()
    List.ofSeq segments

        let trimmed = command.TrimStart()
        let baseCmd =
            if trimmed = "" then ""
            else trimmed.Split([| ' '; '\t' |], 2).[0]
        let withDq, fullyUnquoted, keepQuoteChars = extractQuotedContent command
        let ctx = {
            OriginalCommand        = command
            BaseCommand            = baseCmd
            UnquotedContent        = withDq
            FullyUnquotedContent   = stripSafeRedirections fullyUnquoted
            FullyUnquotedPreStrip  = fullyUnquoted
            UnquotedKeepQuoteChars = keepQuoteChars
        }
        // Steps 3-5: chain
        match runChain ctx with
        | Deny r -> Error r
        | Allow  ->
            // Chain allowed early (empty or safe git commit) — still apply
            // destructive pattern check unconditionally (blueCode runs
            // unattended; destructive patterns always blocked).
            match checkDestructive command with
            | Deny r -> Error r
            | _      -> Ok ()
        | Passthrough ->
            // Chain passed all checks — now verify destructive patterns.
            match checkDestructive command with
            | Deny r -> Error r
            | _      -> Ok ()
```
